Module Text_block.Boxed

Combinators for building up cell structures separated by box characters: e.g.

let a = text "A" in
let b = text "B" in
let c = text "C" in
boxed Boxed.(hcat [vcat [cell a; cell b]; c])

and

let a = text "A" in
let b = text "B" in
let c = text "C" in
boxed Boxed.(hcat [
  vcat [cell a; cell b];
  c;
  vcat [cell b; cell a]
])

produce

┌───┬───┐ ┌───┬───┬───┐ │ A │ │ │ A │ │ B │ ├───┤ C │ and ├───┤ C ├───┤ │ B │ │ │ B │ │ A │ └───┴───┘ └───┴───┴───┘

respectively.
